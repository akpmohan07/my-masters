# 3. Trapezoidal

```c
#include <stdio.h>
#include <omp.h>

#define LOW 0
#define HIGH 10
#define SLICES 1000

// Function to integrate — f(x) = x²
double f(double x) {
    return x * x;
}

// Approximate integral using trapezoidal rule
// Each slice area = (f(start) + f(end)) / 2 * width
double integrate(double low, double high, int slices) {
    double width = (double)(high - low) / slices;
    double total = 0.0;

    // Each iteration is independent — safe to parallelise
    #pragma omp parallel for reduction(+:total)
    for (int i = 0; i < slices; i++) {
        double start = low + i * width;        // left edge of slice
        double end   = start + width;          // right edge of slice
        double area  = (f(start) + f(end)) / 2.0 * width;  // trapezoid area
        total += area;
    }

    return total;
}

int main() {
    double result = integrate(LOW, HIGH, SLICES);
    printf("Result: %f\n", result);   // expected ~333.333
    return 0;
}
```
